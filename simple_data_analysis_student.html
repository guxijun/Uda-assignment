def set_state(BMI):
    if BMI <= 18.5:
        return ''
    elif 18.5 < BMI <= 23.9:
        return ''
    elif 24 < BMI <= 27:
        return ''
    elif 28 < BMI <= 32:
        return ''
    else:
        return ''

data_health['BMI'] = data_health.apply(lambda x : round(x['weight'] / (x['height']/100)**2, 2), axis=1)
data_health['BMI_state'] = data_health.apply(lambda x : set_state(x['BMI']), axis = 1)

plt.figure(figsize = [12, 10])
